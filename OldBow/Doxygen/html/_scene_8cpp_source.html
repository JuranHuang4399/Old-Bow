43void Scene::initEnemies(int numTxComponents) {
44 int row = 2;
45 int col = 18;
46 int enemyIndex = 0; // To keep track of how many enemies have been processed
47 float txComponentX = WINDOW_WIDTH / (col + 2); // plus 2 for extra padding on the left and right edge paddings
48 float txComponentY = WINDOW_HEIGHT / 6; // total 6 rows needed
49
50 movementThreshold = txComponentX;
51
53 for (int i = 1; i <= row; i++) {
54 for(int j = 1; j <= col; j++) {
55 if (enemyIndex >= numTxComponents) {
56 return; // Stop when the maximum of enemies is reached
57 }
58
59 float x = j * txComponentX;
60 float y = i * txComponentY;
61
63 std::shared_ptr<Enemy> enemy = std::make_shared<Enemy> (100.0f, mRenderer);
64 enemy -> init(); // Initialize
66 enemy -> GetTransform() -> move(x, y);
67
69 enemies.push_back(std::move(enemy));
70 enemyIndex++;
71 }
72 }
73
75 if (!enemies.empty()) {
76 movementThreshold -= enemies[0] -> GetTransform() -> getWidth();
77 }
79 Enemy::movementThreshold = movementThreshold;
80}
