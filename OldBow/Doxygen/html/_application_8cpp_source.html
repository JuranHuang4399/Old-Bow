170void Application::loop(int TargetFPS) {
172 Uint32 budgetTime = 1000 / TargetFPS;
173
175 Uint32 startTime, elapsedTime; // Marks start time for a frame (startTime), Stores time spent to finish a frame
176 Uint32 frameEnd = SDL_GetTicks(); // Marks end time when a frame is finished
177 Uint32 lastFPSTime = SDL_GetTicks(); // Marks the last time a minute has passed
185 TimeManager::lastUpdateTime = SDL_GetTicks();
186 int frameCount = 0; // Marks how many frames has been drawn, used to count FPS (frame per second)
187
189 while (currentScene < mScenes.size()) {
191 if (!mScenes[currentScene] -> GetSceneStatus()) {
192 currentScene++;
193 if (currentScene < mScenes.size()) {
194 mScenes[currentScene] -> SceneStartUp(mRenderer);
195 } else {
196 std::cout << "Thank you for playing.\n";
198 break;
199 }
200 }
201
203 startTime = SDL_GetTicks();
204
206 float deltaTime = (startTime - TimeManager::lastUpdateTime) / 1000.0f; // Convert to Seconds by dividing 1000
207 TimeManager::resetLastUpdateTime(); // Reset lastUpdateTime for next frame and prepare for deltaTime calculation
208
213 input(deltaTime);
214
220 if (currentState != GameState::Dialogue) {
221 update(deltaTime);
222 }
223
228 render();
229
231 frameEnd = SDL_GetTicks();
233 elapsedTime = frameEnd - startTime;
234 frameCount++; // One frame has passed
235
237 if (startTime - lastFPSTime >= 1000) {
238 std::string title = "SDL Window - FPS: " + std::to_string(frameCount);
239 SDL_SetWindowTitle(mWindow, title.c_str());
240 frameCount = 0; // Reset frame count after each second
241 lastFPSTime = startTime; // Update lastFPSTime to the current time
242 }
243
245 if (elapsedTime < budgetTime) {
246 SDL_Delay(budgetTime - elapsedTime);
247 }
248 }
