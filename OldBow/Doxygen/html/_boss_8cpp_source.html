99void Boss::loadFromFile (const std::string& filePath) {
100 std::ifstream file(filePath);
101 if (!file.is_open()) {
102 std::cerr << "ERROR: Could not open boss file: " << filePath << std::endl;
103 return;
104 }
105 std::string line;
106 while (std::getline(file, line)) {
112 size_t separatorPos = line.find(":");
113 if (separatorPos == std::string::npos) {
114 continue; // Ignore lines that are not key - value based
115 }
116
118 std::string key = trim(line.substr(0, separatorPos));
119
121 std::string value = trim(line.substr(separatorPos + 1));
122
124 if (key == "ID") {
126 if (value == "Boss1") {
127 mID = BossID::Boss1;
128 } else if (value == "Boss2") {
129 mID = BossID::Boss2;
130 } else if (value == "Boss3") {
131 mID = BossID::Boss3;
132 }
133 } else if (key == "Name") {
135 mName = value;
136 } else if (key == "HP") {
138 mHP = std::stoi(value);
139 mMaxHP = mHP;
140 } else if (key == "ID") {
141
142 } else if (key == "PatternList") {
144 std::stringstream container(value);
146 std::string pattern;
147
149 while (std::getline(container, pattern, ',')) {
150 // Clean up leading spaces with trim()
151 pattern = trim(pattern);
152
154 if (pattern == "Rain") {
155 mPatternList.push_back(PatternType::Rain);
156 } else if (pattern == "Spread") {
157 mPatternList.push_back(PatternType::Spread);
158 } else if (pattern == "Ultimate") {
159 mPatternList.push_back(PatternType::Ultimate);
160 } else if (pattern == "Ghost") {
161 mPatternList.push_back(PatternType::GhostFire);
162 } else if (pattern == "Slash") {
163 mPatternList.push_back(PatternType::Slash);
164 } else {
165 std::cerr << "Failure! Unknown pattern type: " << pattern << std::endl;
166 }
167 }
168 } else if (key == "Texture") {
170 mTexturePath = value;
171 }
172 }
173
175 if (mTexturePath.empty()) {
176 std::cerr << "WARNING! No texture path set for boss!\n";
177 }
178}

187void Boss::update(float deltaTime) {
189 for (auto& component : mComponents) {
190 component.second->update(deltaTime);
191 }
192
194 auto ref = GetTransform();
195 if (!ref) {
196 return;
197 }
198
200 mMovement = mSpeed * deltaTime;
201 float x = ref -> getX();
202 float y = ref -> getY();
203
205 mDirectionChangeTimer -= deltaTime;
206 if (mDirectionChangeTimer <= 0.0f) {
207 if (flipChance(gen) < 50) { // 50 percent chance to flip
208 mXDirection *= -1.0f;
209 } else if (flipChance(gen) < 50 ) { // 50 percent chance to flip
210 mYDirection *= -1.0f;
211 }
212 mDirectionChangeTimer = directionChangeInterval(gen);
213 }
214
215
217 float nextX = x + mMovement * mXDirection;
218 float nextY = y;
219
221 if (mID == BossID::Boss1) {
222 nextY = y + mMovement * mYDirection;
223 if (nextY > 200.0f) {
224 nextY = 200.0f;
225 mYDirection = -1.0f;
226 }
227 if (nextY < 0.0f) {
228 nextY = 0.0f;
229 mYDirection = 1.0f;
230 }
231 }
232
234 if (nextX > 1500.0f) {
235 nextX = 1500.0f;
236 mXDirection = -1.0f;
237 }
238 if (nextX < 400.0f) {
239 nextX = 400.0f;
240 mXDirection = 1.0f;
241 }
242
244 ref -> move(nextX, nextY);
245
250 mRainCooldown -= deltaTime;
251 mSpreadCooldown -= deltaTime;
252 mUltimateTimer -= deltaTime;
253 mGhostFireCooldown -= deltaTime;
254 mSlashCooldown -= deltaTime;
255
257 if (mPatternList.empty()) {
258 return;
259 }
260
261
263 for (PatternType pattern : mPatternList) {
264 switch (pattern) {
265
266 case PatternType::Rain:
267 if (mRainCooldown <= 0.0f) {
268 fireStraightRain();
269 mRainCooldown = mRainFireRate;
270 }
271 break;
272
273 case PatternType::Spread:
274 if (mSpreadCooldown <= 0.0f) {
275 fireSpreadShot();
276 mSpreadCooldown = mSpreadFireRate;
277 }
278 break;
279
281 case PatternType::Ultimate:
282 if (mUltimateTimer <= 0.0f) {
283 dragonBreath();
284 mUltimateFirePeriodTimer -= deltaTime;
285 if (mUltimateFirePeriodTimer <= 0.0f) {
286 mUltimateTimer = mUltimateCooldown;
287 resetUltFirePeriod();
288 }
289 };
290 break;
291
292 case PatternType::GhostFire:
293 if (mGhostFireCooldown <= 0) {
294 fireLingeringGhostShot();
295 mGhostFireCooldown = mGhostFireRate;
296 }
297 break;
298
299 case PatternType::Slash:
300 if (mSlashCooldown <= 0) {
301 slash();
302 mSlashCooldown = mSlashFireRate;
303 }
304 }
305 }
306
307
308
310 for (auto& proj : mProjectiles) {
311 proj -> update(deltaTime);
312 }
313
315 for (auto it = mProjectiles.begin(); it != mProjectiles.end();) {
316 if (!(*it)->getRenderable()) {
317 it = mProjectiles.erase(it);
318 } else {
319 ++it;
320 }
321 }
322}
